# Q2. Largest Component Size by Common Factor
## 접근
각 노드의 루트노드끼리 gcd가 같다면 union 후 부모의 count 세기
```
class Solution {
    var parents = [Int: Int]()
    var count = [Int: Int]()

    func largestComponentSize(_ nums: [Int]) -> Int {
        var answer = 0

        for num in nums {
            parents[num, default: num] = num
            count[num, default: 0] = 1
        }
        for i in 0..<nums.count-1 {
            union(nums[i], nums[i+1])
        }
        for i in 0..<nums.count-1 {
            union(nums[i], nums[i+1])
        }

        for num in nums {
            let countMax = count[num, default: 0]
            answer = answer < countMax ? countMax : answer
        }

        print(count)
        print(parents)

        return answer
    }

    @inline(__always)
    private func gcd(_ a: Int, _ b: Int) -> Int {
        var a = abs(a)
        var b = abs(b)

        while b != 0 {
            let temp = a % b
            a = b
            b = temp
        }

        return a
    }

    private func find( _ target: Int) -> Int {
        guard let parent = parents[target] else { return Int.max }

        if parent == target {
            return target
        }

        let maxParent = find(parent)
        parents[target] = maxParent

        return maxParent
    }

    private func union(_ g: Int, _ v: Int) {
        var g = find(g)
        var v = find(v)

        guard g != Int.max else { return }
        guard v != Int.max else { return }
        guard gcd(g, v) != 1 else { return }
        guard gcd(g, v) != 0 else { return }
        guard g != v else { return }

        if g < v {
            swap(&g, &v)
        }

        count[g]! += count[v]!
        parents[v] = g
    }
}
```

## 실패
루트노드끼리의 gcd 비교는 모든 공통 인수 정보가 없음
따라서 union이 일어나지 않는 경우가 있음
또한 인접한 노드만 union하고있어 같은 컴포넌트에 속해야 하는데 그렇지 못하는 경우가 있음

## 정답
각 노드의 인수를 가상의 노드로 취급하고 union한다.
이후 num에 속하는 것만 검사하여 컴포넌트 내 실제 개수만을 센다.
